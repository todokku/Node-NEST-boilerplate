    return this.cityModel.create(city);
  }

  update(id: string, city: ICity) {
    return this.cityModel.findOneAndUpdate(
      { _id: id },
      { $set: city },
      { new: true },
    );
  }

  getById(id) {
    return this.cityModel.findOne({ _id: id });
  }

  delete(id) {
    return this.cityModel.delete({ _id: id });
  }

  getAll() {
    return this.cityModel.find();
